"use client";
import { useRootData } from "@/src/context/rootContext";
import Image from "next/image";
import Link from "next/link";
import { useRouter } from "next/navigation";
import { useEffect, useRef, useState } from "react";
import {
  HiOutlineCalendar,
  HiOutlineCollection,
  HiOutlineDatabase,
  HiOutlinePhotograph,
} from "react-icons/hi";
import { IoCloseOutline } from "react-icons/io5";
import { LuUserCheck } from "react-icons/lu";
import { MdOutlineError, MdOutlineWarning, MdRefresh } from "react-icons/md";
import { RiLogoutCircleRLine } from "react-icons/ri";
import { VscAccount } from "react-icons/vsc";

// Storage Info Component
const StorageInfo = ({ eventStatistics, onClose, maxStorage }) => {
  const [isMinimized, setIsMinimized] = useState(false);
  const { refreshPackage, isRefreshingPackage } = useRootData();
  const usedPercentage = (eventStatistics?.totalMBUsed / maxStorage) * 100;

  const getStorageStatus = () => {
    if (usedPercentage >= 100)
      return {
        level: "full",
        color: "text-red-600",
        bgColor: "bg-red-50",
        icon: MdOutlineError,
      };
    if (usedPercentage >= 90)
      return {
        level: "critical",
        color: "text-red-500",
        bgColor: "bg-red-50",
        icon: MdOutlineWarning,
      };
    if (usedPercentage >= 80)
      return {
        level: "warning",
        color: "text-yellow-600",
        bgColor: "bg-yellow-50",
        icon: MdOutlineWarning,
      };
    return {
      level: "good",
      color: "text-green-600",
      bgColor: "bg-green-50",
      icon: HiOutlineDatabase,
    };
  };

  const status = getStorageStatus();
  const StatusIcon = status.icon;

  const getWarningMessage = () => {
    if (usedPercentage >= 100)
      return "Storage is full! Please delete some files.";
    if (usedPercentage >= 90)
      return "Storage is almost full! Consider cleaning up.";
    if (usedPercentage >= 80)
      return "Storage usage is high. Monitor carefully.";
    return null;
  };

  const warningMessage = getWarningMessage();

  if (isMinimized) {
    return (
      <div className="fixed bottom-4 right-4 z-40">
        <button
          onClick={() => setIsMinimized(false)}
          className="flex items-center gap-2 px-3 py-2 bg-white rounded-lg shadow-lg border border-gray-200 hover:shadow-xl transition-all duration-300"
          style={{ borderColor: "var(--color-secondary)" }}
        >
          <StatusIcon className={`text-lg ${status.color}`} />
          <span
            className="text-sm font-medium"
            style={{ color: "var(--color-primary)" }}
          >
            {Math.round(usedPercentage)}%
          </span>
        </button>
      </div>
    );
  }

  return (
    <div className="fixed bottom-4 right-4 z-40 w-full max-w-max sm:max-w-md">
      <div className="bg-white rounded-xl shadow-lg border border-gray-200 overflow-hidden">
        {/* Header */}
        <div
          className="flex items-center justify-between p-4 border-b border-gray-100"
          style={{ backgroundColor: "#f8f9fa" }}
        >
          <div className="flex items-center gap-2">
            <HiOutlineDatabase className="text-xl text-primary" />
            <h3 className="font-semibold text-sm text-primary">Storage Info</h3>
          </div>
          <div className="flex items-center gap-2">
            <button
              onClick={refreshPackage}
              disabled={isRefreshingPackage}
              className="p-1 hover:bg-gray-200 rounded-full transition-colors duration-200 disabled:opacity-50"
              title="Refresh Storage Data"
            >
              <MdRefresh
                className={`text-lg text-primary ${
                  isRefreshingPackage ? "animate-spin" : ""
                }`}
              />
            </button>
            <button
              onClick={() => setIsMinimized(true)}
              className="p-1 hover:bg-gray-200 rounded-full transition-colors duration-200"
              title="Minimize"
            >
              <span className="text-sm font-bold text-primary">âˆ’</span>
            </button>
            <button
              onClick={onClose}
              className="p-1 hover:bg-gray-200 rounded-full transition-colors duration-200"
              title="Close"
            >
              <IoCloseOutline className="text-lg text-primary" />
            </button>
          </div>
        </div>

        {/* Warning Message */}
        {warningMessage && (
          <div className={`p-3 border-b border-gray-100 ${status.bgColor}`}>
            <div className="flex items-center gap-2">
              <StatusIcon className={`text-lg ${status.color}`} />
              <p className={`text-xs font-medium ${status.color}`}>
                {warningMessage}
              </p>
            </div>
          </div>
        )}

        {/* Storage Usage Bar */}
        <div className="p-4 border-b border-gray-100">
          <div className="flex justify-between items-center mb-2">
            <span className="text-xs font-medium text-primary">
              Storage Used
            </span>
            <span className="text-xs font-bold text-primary">
              {eventStatistics?.totalMBUsed}MB / {maxStorage}MB
            </span>
          </div>
          <div className="w-full bg-gray-200 rounded-full h-2">
            <div
              className="h-2 rounded-full transition-all duration-300"
              style={{
                width: `${Math.min(usedPercentage, 100)}%`,
                backgroundColor:
                  usedPercentage >= 90
                    ? "#ef4444"
                    : usedPercentage >= 80
                      ? "#f59e0b"
                      : "var(--color-secondary)",
              }}
            />
          </div>
          <div className="text-xs text-gray-500 mt-1">
            {Math.round(usedPercentage)}% used
          </div>
        </div>

        {/* Statistics Grid */}
        <div className="p-4">
          <div className="grid grid-cols-2 gap-3">
            <div
              className="flex items-center gap-2 p-2 rounded-lg"
              style={{ backgroundColor: "#f8f9fa" }}
            >
              <HiOutlineCalendar className="text-lg text-secondary" />
              <div>
                <p className="text-xs text-gray-500">Events</p>
                <p className="text-sm font-semibold text-primary">
                  {eventStatistics?.totalEvents}
                </p>
              </div>
            </div>

            <div
              className="flex items-center gap-2 p-2 rounded-lg"
              style={{ backgroundColor: "#f8f9fa" }}
            >
              <HiOutlinePhotograph className="text-lg text-secondary" />
              <div>
                <p className="text-xs text-gray-500">Photos</p>
                <p className="text-sm font-semibold text-primary">
                  {eventStatistics?.totalPhotos}
                </p>
              </div>
            </div>

            <div
              className="flex items-center gap-2 p-2 rounded-lg"
              style={{ backgroundColor: "#f8f9fa" }}
            >
              <HiOutlineCollection className="text-lg text-secondary" />
              <div>
                <p className="text-xs text-gray-500">Albums</p>
                <p className="text-sm font-semibold text-primary">
                  {eventStatistics?.totalAlbums}
                </p>
              </div>
            </div>

            <div
              className="flex items-center gap-2 p-2 rounded-lg"
              style={{ backgroundColor: "#f8f9fa" }}
            >
              <HiOutlineDatabase className="text-lg text-secondary" />
              <div>
                <p className="text-xs text-gray-500">Storage</p>
                <p className="text-sm font-semibold text-primary">
                  {eventStatistics?.totalMBUsed}MB
                </p>
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
};

const Navbar = () => {
  const router = useRouter();
  const [isPopupOpen, setIsPopupOpen] = useState(false);
  const [showStorageInfo, setShowStorageInfo] = useState(false);
  const modalRef = useRef(null);
  const { user, onePackage, refreshPackage, isRefreshingPackage } =
    useRootData();
  const eventStatistics = onePackage?.eventStatistics;
  const handleLogout = async () => {
    try {
      const { logOutUser } =
        await import("@/src/redux/features/auth/authThunks");
      await logOutUser(router, {});
    } catch (error) {
      console.error("Error saving data:", error);
    }
  };

  useEffect(() => {
    const handleClickOutside = (event) => {
      if (modalRef.current && !modalRef.current.contains(event.target)) {
        setIsPopupOpen(false);
      }
    };
    document.addEventListener("mousedown", handleClickOutside);
    return () => {
      document.removeEventListener("mousedown", handleClickOutside);
    };
  }, [isPopupOpen]);
  // Calculate storage usage percentage using real package limit
  const maxStorage = onePackage?.packageId?.totalPhotoMB;
  const usedPercentage =
    maxStorage > 0 ? (eventStatistics?.totalMBUsed / maxStorage) * 100 : 0;
  const getStorageIndicatorColor = () => {
    if (usedPercentage >= 90) return "text-red-500";
    if (usedPercentage >= 80) return "text-yellow-500";
    return "text-green-500";
  };

  return (
    <>
      <nav className="flex justify-between items-start sm:items-center mb-2.5 w-auto mt-1">
        <div className="mb-3 sm:mb-0">
          <div className="flex items-center gap-2">
            <h1 className="text-primary md:text-lg lg:text-2xl font-bold">
              Hello {user?.fullName}! ðŸ‘‹
            </h1>
            <div className="w-2 h-2 rounded-full bg-green-500 animate-pulse"></div>
          </div>
          <div className="flex flex-col sm:flex-row sm:items-center gap-1 sm:gap-4">
            <p className="text-xs md:text-sm text-gray-500">
              Welcome to your FindMyPhotos dashboard...
            </p>
            <div className="flex items-center gap-2 text-xs md:text-sm">
              <span className="text-gray-500">Storage:</span>
              <span className="font-medium text-primary">
                {eventStatistics?.totalMBUsed}MB /{" "}
                {onePackage?.packageId?.totalPhotoMB}MB
              </span>
              <span className="text-gray-400">
                ({onePackage?.packageId?.packageName}
                {onePackage?.packageId?.isTrial ? " - Trial" : ""})
              </span>
            </div>
          </div>
        </div>

        <div className="flex sm:flex-row flex-col items-center gap-2 relative">
          {/* Storage Info Button */}
          <button
            onClick={() => setShowStorageInfo(true)}
            className="flex items-center gap-2 sm:p-2 sm:hover:bg-gray-50 rounded-full transition-all duration-300 hover:shadow-sm group sm:order-none order-2"
            title="Storage Info"
          >
            <div className="relative">
              <HiOutlineDatabase
                className={`text-2xl ${getStorageIndicatorColor()} transition-colors duration-300`}
              />
              {usedPercentage >= 80 && (
                <div className="absolute -top-1 -right-1 w-3 h-3 bg-red-500 rounded-full border border-white animate-pulse"></div>
              )}
            </div>
            <span className="text-xs font-medium text-gray-700 group-hover:text-primary transition-colors duration-300">
              {Math.round(usedPercentage)}%
            </span>
          </button>
          <button
            onClick={refreshPackage}
            disabled={isRefreshingPackage}
            className="p-1 hover:bg-gray-200 rounded-full transition-colors duration-200 disabled:opacity-50"
            title="Refresh Storage Data"
          >
            <MdRefresh
              className={`text-lg text-primary ${
                isRefreshingPackage ? "animate-spin" : ""
              }`}
            />
          </button>

          {/* User Profile Button */}
          <button
            onClick={() => setIsPopupOpen(!isPopupOpen)}
            className="flex items-center gap-2 sm:p-2 sm:hover:bg-gray-50 rounded-full transition-all duration-300 hover:shadow-sm group"
          >
            {user?.image ? (
              <Image
                src={user.image}
                alt="Profile"
                width={64}
                height={64}
                className="w-8 h-8 rounded-xl object-cover shadow-md"
              />
            ) : (
              <div className="w-8 h-8 bg-gradient-to-br from-secondary to-primary rounded-xl flex items-center justify-center text-white text-xl font-bold shadow-md">
                {user?.fullName?.[0]?.toUpperCase() || "F"}
              </div>
            )}
            <span className="hidden xl:inline text-sm font-medium text-gray-700 group-hover:text-primary transition-colors duration-300">
              {user?.fullName}
            </span>
          </button>

          {/* User Popup Menu */}
          {isPopupOpen && (
            <div
              ref={modalRef}
              className="absolute z-50 top-full right-0 mt-2 w-64 bg-white rounded-xl shadow-lg border border-gray-100 overflow-hidden animate-fadeIn"
            >
              <div className="absolute -top-2 right-4 w-4 h-4 bg-white transform rotate-45 border-t border-l border-gray-100"></div>
              <div className="p-4 border-b border-gray-100">
                <div className="flex items-center gap-3">
                  {user?.image ? (
                    <Image
                      src={user.image}
                      alt="Profile"
                      width={64}
                      height={64}
                      className="w-8 h-8 rounded-xl object-cover shadow-md"
                    />
                  ) : (
                    <div className="w-8 h-8 bg-gradient-to-br from-secondary to-primary rounded-xl flex items-center justify-center text-white text-xl font-bold shadow-md">
                      {user?.fullName?.[0]?.toUpperCase() || "F"}
                    </div>
                  )}
                  <div>
                    <p className="font-medium text-gray-900">
                      {user?.fullName}
                    </p>
                    <p className="text-xs text-gray-500">Online</p>
                  </div>
                </div>
              </div>
              <ul className="py-1">
                <li>
                  <Link
                    href="/my-account"
                    className="flex items-center px-4 py-3 text-sm text-gray-700 hover:bg-gray-50 transition-colors duration-150 group"
                  >
                    <LuUserCheck className="text-xl text-secondary mr-3 group-hover:scale-110 transition-transform duration-200" />
                    <span>My Account</span>
                  </Link>
                </li>
                <li className="border-t border-gray-100">
                  <button
                    className="flex items-center w-full px-4 py-3 text-sm text-red-600 hover:bg-red-50 transition-colors duration-150 group"
                    onClick={handleLogout}
                    type="button"
                  >
                    <RiLogoutCircleRLine className="text-xl text-red-600 mr-3 group-hover:scale-110 transition-transform duration-200" />
                    <span>Logout</span>
                  </button>
                </li>
              </ul>
            </div>
          )}
        </div>
      </nav>

      {/* Storage Info Modal */}
      {showStorageInfo && (
        <StorageInfo
          eventStatistics={eventStatistics}
          onClose={() => setShowStorageInfo(false)}
          maxStorage={maxStorage}
        />
      )}
    </>
  );
};

export default Navbar;
